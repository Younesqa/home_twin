Example:
function buildRoomLayouts(rooms) {
  const layoutMap = new Map();

  const living = rooms.find(r => r.type === "livingRoom");
  const kitchen = rooms.find(r => r.type === "kitchen");
  const bathroom = rooms.find(r => r.type === "bathroom");
  const bedrooms = rooms.filter(r => r.type === "bedroom");

  if (living) layoutMap.set(living.id, { x: 40, y: 120, w: 250, h: 150 });
  if (kitchen) layoutMap.set(kitchen.id, { x: 310, y: 120, w: 250, h: 150 });
  if (bathroom) layoutMap.set(bathroom.id, { x: 40, y: 290, w: 160, h: 110 });

  const bedroomSlots = [
    { x: 220, y: 290, w: 160, h: 110 },
    { x: 400, y: 290, w: 160, h: 110 },
    { x: 40, y: 420, w: 160, h: 90 },
    { x: 220, y: 420, w: 160, h: 90 },
    { x: 400, y: 420, w: 160, h: 90 },
  ];

  bedrooms.forEach((room, index) => {
    layoutMap.set(room.id, bedroomSlots[index] || bedroomSlots[bedroomSlots.length - 1]);
  });

  return layoutMap;
}
